float mover(float centimetros, int eje) {
  float tiempo = ((0.0598 * centimetros) + 0.3235) * 1000;

  if (eje == 3) {
    motor1(100, 0);
    motor2(100, 1);
    delay(tiempo);
    motor1(0, 0);
    motor2(0, 1);
  } 
  else if (eje == 1) {
    motor3(100, 0);
    motor2(100, 1);
    delay(tiempo);
    motor3(0, 0);
    motor2(0, 1);
  } 
  else if (eje == 2) {
    motor3(100, 0);
    motor1(100, 1);
    delay(tiempo);
    motor3(0, 0);
    motor1(0, 1);
  }

  return tiempo;
}

float girar(float angulo) {
  float tiempo = angulo * 300 / 160.0;

  motor1(100, 1);
  motor2(100, 1);
  motor3(100, 1);
  delay(tiempo);
  motor1(0, 1);
  motor2(0, 1);
  motor3(0, 1);

  return tiempo;
}
